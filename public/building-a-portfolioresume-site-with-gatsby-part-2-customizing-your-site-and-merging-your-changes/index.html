67        });

69        // Prevent rerun of the listener attachment
70        setHasCalculated(true);
71      }
72

74      setContentHeight(contentSection.getBoundingClientRect().height);
75    }, 20);

78    window.addEventListener("resize", calculateBodySize);

81  }, []);
